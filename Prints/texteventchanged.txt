package pract13;

import java.awt.*;
import java.awt.event.*;

import javax.swing.*;
import javax.swing.JLabel;

public class TextListnerDemo extends Frame implements TextListener{
	 TextField t1,t2; 
	 Button b;
	public TextListnerDemo() {

		setSize (400, 400);  
		setLayout(new FlowLayout());   
		setVisible(true); 
		t1 = new TextField(30);
		t2 = new TextField(30);
		add(t1); 
		add(t2);  
		t1.addTextListener(this);
	}

	public static void main(String[] args) {
		new TextListnerDemo();
	}

	@Override
	public void textValueChanged(TextEvent e) {
		t2.setText("Welcome");
	}

}
